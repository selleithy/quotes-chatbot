    <script>

        const API_BASE = 'http://localhost:3000';


        function addMessage(text, isUser = false) {

            const chatContainer = document.getElementById('chatContainer');

            const messageDiv = document.createElement('div');

            messageDiv.className = `message ${isUser ? 'user-message' : 'bot-message'}`;

            messageDiv.innerHTML = isUser ? `<div>${text}</div>` : `

                <div>${text}</div>

                <div class="quote" id="quote"></div>

                <div class="author" id="author"></div>

            `;

            if (!isUser) {

                // For bot, we'll fill quote and author after fetch

                messageDiv.querySelector('.quote').id = 'tempQuote';

                messageDiv.querySelector('.author').id = 'tempAuthor';

            }

            chatContainer.appendChild(messageDiv);

            chatContainer.scrollTop = chatContainer.scrollHeight;

        }


        async function getUpliftingQuote(feeling) {

            try {

                const response = await fetch(`${API_BASE}/quotes/by-feeling?feeling=${feeling}`);

                const quote = await response.json();

                return quote;

            } catch (error) {

                console.error('Error loading quote:', error);

                return { text: 'Hang in there! Sometimes the best quotes come from within.', author: 'You' };

            }

        }


        async function selectFeeling(feeling) {

            const userMessage = `I'm feeling ${feeling}.`;

            addMessage(userMessage, true);

            const quote = await getUpliftingQuote(feeling);

            setTimeout(() => {

                const tempQuote = document.getElementById('tempQuote');

                const tempAuthor = document.getElementById('tempAuthor');

                if (tempQuote) tempQuote.textContent = quote.text;

                if (tempAuthor) tempAuthor.textContent = `- ${quote.author}`;

            }, 500);

            addMessage(`Here's something to uplift you for feeling ${feeling}:`, false);

        }


        async function sendMessage() {

            const input = document.getElementById('userInput');

            const message = input.value.trim();

            if (!message) return;

            addMessage(message, true);

            input.value = '';


            // Simple keyword matching for feelings

            let feeling = 'general';

            const lowerMessage = message.toLowerCase();

            if (lowerMessage.includes('sad') || lowerMessage.includes('down')) feeling = 'sad';

            else if (lowerMessage.includes('anxious') || lowerMessage.includes('worry')) feeling = 'anxious';

            else if (lowerMessage.includes('stress') || lowerMessage.includes('overwhelmed')) feeling = 'stressed';

            else if (lowerMessage.includes('unmotivated') || lowerMessage.includes('tired')) feeling = 'unmotivated';


            const quote = await getUpliftingQuote(feeling);

            addMessage(`I sense you're feeling ${feeling}. Here's an uplifting quote:`, false);

            setTimeout(() => {

                const tempQuote = document.getElementById('tempQuote');

                const tempAuthor = document.getElementById('tempAuthor');

                if (tempQuote) tempQuote.textContent = quote.text;

                if (tempAuthor) tempAuthor.textContent = `- ${quote.author}`;

            }, 500);

        }


        function handleKeyPress(event) {

            if (event.key === 'Enter') sendMessage();

        }


        async function addQuote(event) {

            event.preventDefault();

            const text = document.getElementById('textInput').value;

            const author = document.getElementById('authorInput').value || 'Unknown';

            const feeling = document.getElementById('feelingInput').value;


            try {

                const response = await fetch(`${API_BASE}/quotes`, {

                    method: 'POST',

                    headers: { 'Content-Type': 'application/json' },

                    body: JSON.stringify({ text, author, feeling })

                });

                if (response.ok) {

                    alert('Quote added successfully!');

                    document.getElementById('textInput').value = '';

                    document.getElementById('authorInput').value = '';

                    document.getElementById('feelingInput').value = '';

                } else {

                    alert('Error adding quote');

                }

            } catch (error) {

                console.error('Error adding quote:', error);

                alert('Error adding quote');

            }

        }
